\documentclass{article}
\usepackage{etoolbox}

% Initialize the list as empty
\newcommand{\mylist}{}

% Define a command to append items to the list
\newcommand{\appendtomylist}[1]{%
  \ifdefempty{\mylist}
    {\def\mylist{#1}} % If mylist is empty, simply define it
    {\edef\mylist{\mylist,#1}} % Otherwise, append the new item with a preceding comma
}

% Define a command to process and print each item in the list individually
\newcommand{\processmylist}{%
  \renewcommand*\do[1]{%
    % This is where each item is actually processed
    Item: ##1\par
  }
  \expandafter\docsvlist\expandafter{\mylist}
}

\begin{document}

\appendtomylist{First item}
\appendtomylist{Second item}
\appendtomylist{Third item}

\section*{Items in My List}
\processmylist

\end{document}
